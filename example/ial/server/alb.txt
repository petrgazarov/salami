@uses(CumuliVpc)
Resource type: ec2.SecurityGroup
Logical name: ALBSecurityGroup
Name: cumuli-alb-security-group
Description: Security group for ALB
@property(Has open egress rules for all traffic)
@property(Has open ingress rules for all traffic on port 443)

@uses(PublicSubnetA, PublicSubnetB)
Resource type: lb.LoadBalancer
Logical name: CumuliServerAlb
Name: cumuli-server-alb
Internal: False
Security group: ALBSecurityGroup
@property(Deletion protection enabled)
Idle timeout: 3600

@uses(CumuliVpc)
Resource type: lb.TargetGroup
Logical name: CumuliServerTargetGroup
Name: cumuli-server-target-group
Port: 80
Protocol: HTTP
Target type: IP
Health check: every 30 seconds on path "/" using the HTTP protocol on the traffic port
Stickiness: A configuration that enables stickiness with a type of "lb_cookie" and a cookie duration of 1 day

@uses(CumuliServerAlb, CumuliServerCertificate)
Resource type: lb.Listener
Logical name: CumuliServerListener
port: 443
protocol: HTTPS
SSL policy: ELBSecurityPolicy-TLS13-1-2-2021-06
Default actions: A list containing a single action to "forward" incoming requests to CumuliServerTargetGroup.

@uses(CumuliServerAlb, CumuliHostedZone)
Resource type: route53.Record
Logical name: AppALBRecord
name: app.{domain_name}
type: A
aliases: A list containing a single alias. The alias has "name" and "zone_id" of CumuliServerAlb, and "evaluate_target_health" set to True
